Frontend:
    main

heat_balancer:
    /add() -> (id)
    /remove(id)
    /start(id)
    /stop(id)
    /status() -> ([data])

Heater:
    /start
    /stop
    /status
    /ok

/add frontend -> heat_balancer
heater: {
    name: string,
    id: int,
    ip: string,
}

/status heater -> heat_balancer
enum state: {
    pause,
    running,
}

/status heat_balancer -> frontend
data: [
    {
        id: int,
        status: state,
    }
]

/ok heater -> heat_balancer 
enum connection: {
    connected,
    failed,
}
